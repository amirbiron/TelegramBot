👋למה המדריך הזה?

בניית בוט טלגרם זה קל. בניית בוט יעיל, יציב ומקצועי שמשרת אלפי משתמשים - זה אתגר אחר לגמרי.

המדריך הזה מבוסס על לקחים מעשיים מפיתוח CodeBot - בוט לניהול קוד עם:

📊 Coverage גבוה

כיסוי טסטים מעל 60% עם אינטגרציה ל-Codecov

🔄 CI/CD מתקדם

GitHub Actions עם בדיקות אוטומטיות ומיזוג חכם

📈 Monitoring מובנה

Structured logging, Sentry, Prometheus metrics

🛡️ אבטחה מתקדמת

Secret scanning, encrypted tokens, rate limiting

💡 העיקרון המרכזי: הקוד עצמו הוא רק 30% מהעבודה. 70% האחרים הם תשתיות, ניטור, בדיקות, תיעוד ותחזוקה.

🏗️ מבנה שכבות (Layered Architecture)

הפרד את הקוד לשכבות ברורות, כל אחת עם אחריות מוגדרת:

שכבות מומלצות
שכבה תפקיד דוגמה
Handlers קבלת פקודות מהמשתמש bot_handlers.py
Services לוגיקה עסקית services/code_service.py
Repository גישה לנתונים database/repository.py
Models הגדרת מבני נתונים database/models.py
Utils פונקציות עזר utils.py

דוגמה: זרימת בקשת שמירת קובץ

# handlers/save_flow.py
async def get_code(update, context):
    code = update.message.text
    context.user_data['code_to_save'] = code
    return GET_FILENAME

async def get_filename(update, context):
    filename = update.message.text.strip()
    user_id = update.message.from_user.id
    
    # קריאה לשכבת Service
    detected_lang = code_service.detect_language(code, filename)
    
    # קריאה לשכבת Repository
    snippet = CodeSnippet(
        user_id=user_id,
        file_name=filename,
        code=code,
        programming_language=detected_lang
    )
    success = db.save_code_snippet(snippet)
    
    if success:
        await update.message.reply_text("✅ נשמר בהצלחה!")
    return ConversationHandler.END

        IndexModel([("file_name", ASCENDING)]),
        IndexModel([("created_at", DESCENDING)]),

        IndexModel([

            ("code", TEXT),
            ("file_name", TEXT),

        IndexModel(
            [("deleted_expires_at", ASCENDING)],
            name="deleted_ttl",

    self.collection.create_indexes(indexes)

⚡ Redis Cache

Cache חכם מפחית עומס על מסד הנתונים ומשפר זמני תגובה:

# cache_manager.py
from functools import wraps
import redis

redis_client = redis.Redis(
    host='localhost',
    port=6379,
    decode_responses=True
)

def cached(expire_seconds=300, key_prefix=""):
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            # בניית מפתח cache
            cache_key = f"{key_prefix}:{func.__name__}:{args}:{kwargs}"
            
            # בדיקת cache
            cached_value = redis_client.get(cache_key)
            if cached_value:
                return json.loads(cached_value)
            
            # קריאה לפונקציה
            result = await func(*args, **kwargs)
            
            # שמירה ב-cache
            redis_client.setex(
                cache_key,
                expire_seconds,
                json.dumps(result)
            )
            return result
        return wrapper
    return decorator

# שימוש
@cached(expire_seconds=300)
async def get_user_files(user_id: int):
    return db.find({"user_id": user_id}).to_list()

מה כדאי ל-Cache?

פעולה TTL מומלץ סיבה
רשימת קבצים 5 דקות נקרא הרבה, משתנה לעיתים רחוקות
סטטיסטיקות משתמש 1 שעה חישוב כבד, לא קריטי שיהיה real-time
קובץ אחרון שנשמר 30 שניות משתנה לעיתים קרובות
קונפיגורציה 24 שעות כמעט לא משתנה

            "is_active": True
        })
    
    def get_file(self, user_id, file_name):

# observability.py
def _redact_sensitive(logger, method, event_dict):
    """הסרת מידע רגיש מהלוגים"""
    sensitive_keys = {
        "token", "password", "secret", 
        "authorization", "cookie"
    }
    
    for key in list(event_dict.keys()):
        if any(s in key.lower() for s in sensitive_keys):
            event_dict[key] = "[REDACTED]"
    
    return event_dict

הצפנת Tokens

# secret_manager.py
from cryptography.fernet import Fernet

class SecretManager:
    def __init__(self, encryption_key: str):
        self.cipher = Fernet(encryption_key.encode())
    
    def encrypt(self, plaintext: str) -> str:
        """הצפנת טוקן"""
        return self.cipher.encrypt(plaintext.encode()).decode()
    
    def decrypt(self, ciphertext: str) -> str:
        """פענוח טוקן"""
        return self.cipher.decrypt(ciphertext.encode()).decode()

# שימוש

# rate_limiter.py
from collections import defaultdict
from datetime import datetime, timedelta

class RateLimiter:
    def __init__(self):
        self.requests = defaultdict(list)
    
    async def check_limit(
        self, 
        user_id: int, 
        max_per_minute: int = 30
    ) -> bool:
        """בדיקה: האם משתמש חרג מהמגבלה"""
        now = datetime.now()
        one_min_ago = now - timedelta(minutes=1)
        
        # נקה requests ישנים
        self.requests[user_id] = [
            ts for ts in self.requests[user_id]
            if ts > one_min_ago
        ]
        
        # בדוק limit
        if len(self.requests[user_id]) >= max_per_minute:
            return False  # חרג!
        
        self.requests[user_id].append(now)
        return True

# middleware
async def rate_limit_middleware(update, context):
    user_id = update.effective_user.id

            "⚠️ יותר מדי בקשות! אנא המתן דקה."
        )
        raise ApplicationHandlerStop
    
    return True

🛡️ Input Validation

import re

def validate_filename(filename: str) -> bool:
    """אימות שם קובץ - מניעת path traversal"""
    # רק אותיות, מספרים, נקודות וקווים
    if not re.match(r'^[\w\.\-\_]+\.[a-zA-Z0-9]+$', filename):
        return False
    
    # מניעת path traversal
    if '..' in filename or '/' in filename:
        return False
    
    # הגבלת אורך
    if len(filename) > 255:
        return False
    
    return True

def sanitize_code(code: str) -> str:
    """ניקוי קוד מתווים מסוכנים"""
    # הסרת NULL bytes
    code = code.replace('\x00', '')
    
    # הסרת תווי בקרה (מלבד tab ו-newline)
    code = ''.join(
        c for c in code 
        if c in ('\t', '\n', '\r') or not unicodedata.category(c).startswith('C')
    )
    
    return code

  pull_request:
    types: [opened, synchronize]

  auto-merge:
    if: github.actor == 'dependabot[bot]'
    runs-on: ubuntu-latest
    steps:
      - name: Fetch metadata
        id: meta
        uses: dependabot/fetch-metadata@v2
      
      - name: Auto-approve patch updates
        if: steps.meta.outputs.update-type == 'version-update:semver-patch'
        run: gh pr review --approve "$PR_URL"
        env:
          PR_URL: ${{github.event.pull_request.html_url}}
          GITHUB_TOKEN: ${{secrets.GITHUB_TOKEN}}
      
      - name: Enable auto-merge
        if: steps.meta.outputs.update-type == 'version-update:semver-patch'
        run: gh pr merge --auto --squash "$PR_URL"
